program fibonacci:

  define cont as Integer = 0 .
  define c as Integer = 1 .
  define d as Integer = 1 .
  define temp as Integer = 0 .

  while(cont < 10):
    if(cont <= 1):
      print(1).
    else:
      temp = c + d.
      c = d.
      d = temp.
      print(temp).
    end if
    cont = cont + 1 .
  end while

end program
